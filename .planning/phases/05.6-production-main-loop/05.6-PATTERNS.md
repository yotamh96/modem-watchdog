#    AND C-01..C-05). One STATUS= per cycle at cycle-end.
async def _cycle_loop() -> None:
    cycle_id = 0
    while not shutdown_event.is_set():
        # a. Wait for wake (event_queue OR poll deadline). RESEARCH §3.5
        #    canonical pattern: asyncio.wait([queue.get(), scheduler.deadline]).
        await _wait_wake_or_deadline(event_queue, scheduler)

        # b. Run one cycle.
        result = await driver.run_one_cycle(cycle_id=cycle_id)
        cycle_count_ref[0] = cycle_id + 1  # SigtermChoreography reads this

        # c. Persist status.json (cycle is proven complete).
        status_report = _build_status_report(result, ...)
        write_status_json(state_root_path / "status.json", status_report)

        # d. WATCHDOG=1 fires AFTER status.json — Issue #5 / PITFALLS §4.1.
        sd.watchdog_kick()

        # e. STATUS= per cycle (C-05 exact format).
        healthy = sum(1 for m in status_report.modems if m.state == "healthy")
        sd.status(f"cycle={cycle_id} healthy={healthy}/4 actions={...} drift={...:.1f}s")

        # f. READY=1 ONLY after cycle 0 (C-04 / L-01 / PITFALLS §4.1).
        if cycle_id == 0:
            sd.ready(f"first cycle ok healthy={healthy}/4")

        cycle_id += 1
        scheduler.advance()

# 7) SIGTERM / SIGHUP watcher coroutines.
async def _sigterm_watcher() -> None:
    await shutdown_event.wait()
    # Construct choreography NOW (after producer_tasks list is populated).
    # The choreography's step 1+2 cancel the cycle + producers; the
    # TaskGroup then exits via CancelledError propagation.
    choreography = SigtermChoreography(
        cycle_driver_task=cycle_task,
        producer_tasks=producer_tasks,
        webhook_poster=webhook_poster,
        event_logger=event_logger,
        metrics_socket_path=Path(settings.metrics_socket_path),
        run_dir=run_dir,
        clock=clock,
        boot_monotonic=boot_mono,
        cycle_count_ref=cycle_count_ref,
        state_flush=None,  # cycle driver guarantees atomic per-cycle writes
    )
    await choreography.execute(deadline_seconds=5.0)

async def _sighup_watcher() -> None:
    while not shutdown_event.is_set():
        await sighup_event.wait()
        sighup_event.clear()
        await swapper.try_apply_reload()

# 8) Spawn TaskGroup with 7 tasks (4 producers + cycle + sigterm + sighup).
producer_tasks: list[asyncio.Task[object]] = []
async with asyncio.TaskGroup() as tg:
    # Each producer wrapped in restart_on_crash (Pitfall 15 envelope).
    producer_tasks.append(tg.create_task(restart_on_crash(
        "udev_producer", _udev_factory,
        sleeper=..., event_logger=event_logger, clock=clock,
    )))
    producer_tasks.append(tg.create_task(restart_on_crash(
        "rtnetlink_producer", _rtnetlink_factory,
        sleeper=..., event_logger=event_logger, clock=clock,
    )))
    producer_tasks.append(tg.create_task(restart_on_crash(
        "asyncinotify_producer", _asyncinotify_factory,
        sleeper=..., event_logger=event_logger, clock=clock,
    )))
    producer_tasks.append(tg.create_task(restart_on_crash(
        "kmsg_producer", _kmsg_factory,
        sleeper=..., event_logger=event_logger, clock=clock,
    )))
    cycle_task = tg.create_task(_cycle_loop())  # 5th task
    tg.create_task(_sigterm_watcher())          # 6th task
    tg.create_task(_sighup_watcher())           # 7th task

### Pattern 7: Signal handler installation order (cross-file)

**Source:** `src/spark_modem/daemon/main.py:266-293` (placeholder doc-comment, verbatim).
**Apply to:** `_production_main` TaskGroup wiring.

```python
# Signal handler installation BEFORE TaskGroup (Pattern 7):
loop = asyncio.get_running_loop()
loop.add_signal_handler(signal.SIGTERM, shutdown_event.set)
loop.add_signal_handler(signal.SIGHUP, sighup_event.set)
# NEVER signal.signal() (CLAUDE.md anti-pattern).
```

The `shutdown_event` and `sighup_event` are `asyncio.Event` objects constructed in the enclosing `_production_main` body BEFORE `loop.add_signal_handler` is called. CONTEXT.md Claude's Discretion line "shutdown_event / sighup_event placement — module-scope vs passed-as-arg vs daemon/main.py-local. Planner picks; pattern is asyncio.Event() created inside _production_main body before loop.add_signal_handler calls" — local is the established Phase 3 shape.

### Pattern L-01: sd_notify cadence (cycle-loop body)

### Pattern: RELOAD_DATA Settings field

**Source:** `src/spark_modem/config/settings.py:80-85` (`ladder_min_interval_seconds`), 100-106 (`multi_modem_threshold_fraction`), 134-148 (signal floors).
**Apply to:** `cycle_interval_seconds` (C-02). The pattern is mechanical:

```python
<field_name>: <type> = Field(
    default=<default>,
    <constraint, e.g. ge=>,
    json_schema_extra=RELOAD_DATA,
    description="<spec citation>",
)
```

The SIGHUP swapper (sighup.py:132-144) auto-discovers the field via `model_fields` — no extra wiring needed.

### Pattern: Integration test pytestmark

**Source:** `tests/integration/test_lifecycle.py:63-70`.

pytestmark = [
