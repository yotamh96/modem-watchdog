       """

       import subprocess
       import sys
       from pathlib import Path

       @pytest.mark.skipif(

           events_path.write_text("seed line 1\n", encoding="utf-8")

           conf_path.write_text(
               f"""
               {events_path} {{
                   daily
                   rotate 7
                   missingok
                   notifempty
                   sharedscripts
                   create 0640 {os.environ.get('USER', 'root')} {os.environ.get('USER', 'root')}
                   postrotate
                   endscript
               }}
               """,
               encoding="utf-8",

           #    components; FakeClock for the cycle scheduler only).

           # ... wire up real asyncinotify_producer watching log_dir
           # ... spawn the producer task in a TaskGroup

           # 3. Force logrotate to run NOW.
           result = subprocess.run(  # noqa: PLW1510 — explicit allow per integration tier
               [
                   _LOGROTATE_BIN,
                   "-f",  # force rotation regardless of size/time
                   "-s", str(state_path),
                   str(conf_path),
               ],
               capture_output=True,
               text=True,
               check=False,
               timeout=10.0,
           )
