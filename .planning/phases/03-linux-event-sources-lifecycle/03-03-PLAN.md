       ```python
       """rtnetlink producer — push WakeSignal.RTNETLINK on link-state changes.

       PITFALLS §6.1 PRESCRIPTIVE: tight read loop only. Body MUST be
       `event_queue.put_nowait(WakeSignal.RTNETLINK)` ONLY. NO parsing,
       NO state, NO logging.

       SO_RCVBUF=4MiB is required — kernel default 256KiB is too small
       to absorb a Tegra USB hub PSU droop's re-enumeration storm
       (16+ events in 2s). On ENOBUFS the OSError escapes this
       coroutine; the supervisor (event_sources/supervisor.py
       restart_on_crash wrapper) catches Exception and restarts the
       factory with bounded backoff (PITFALLS §6.1: close+reopen +
       force inventory refresh).

       NEVER sync IPRoute (run_in_executor would block the loop;
       anti-pattern catalogue).
       """
       from __future__ import annotations

       import logging
       import socket
       from typing import Protocol

       from spark_modem.event_sources.supervisor import WakeSignal

       logger = logging.getLogger(__name__)


       class _EventQueueProto(Protocol):
           def put_nowait(self, item: object) -> None: ...


       class _AsyncIPRouteProto(Protocol):
           """Subset of pyroute2.AsyncIPRoute the producer touches.

           Co-located so tests inject FakeAsyncIPRoute without
           monkeypatching pyroute2.
           """
           async def __aenter__(self) -> "_AsyncIPRouteProto": ...
           async def __aexit__(self, *exc: object) -> None: ...
           async def bind(self, *, groups: int) -> None: ...
           def get(self) -> object: ...  # async iterator
           # asyncore.socket exposes setsockopt; co-locate just enough.


       def _build_default_ipr() -> _AsyncIPRouteProto:
           """Construct a real pyroute2.AsyncIPRoute. Linux-only."""
           from pyroute2 import AsyncIPRoute  # type: ignore[import-untyped]
           return AsyncIPRoute()


       async def run_rtnetlink_producer(
           *,
           event_queue: _EventQueueProto,
           ipr_factory: object | None = None,
       ) -> None:
           """Subscribe to RTNETLINK link-state changes; push WakeSignal.RTNETLINK.

           Tight read loop only — body is event_queue.put_nowait(...) ONLY.
           ENOBUFS escapes; supervisor restarts.
           """
           # Defer pyroute2 + rtnl imports so module-import works on Windows.
           if ipr_factory is None:
               from pyroute2.netlink import rtnl  # type: ignore[import-untyped]
               groups = rtnl.RTMGRP_LINK
               ipr_cm = _build_default_ipr()
           else:
               # Tests inject a (factory_callable, groups_constant) tuple.
               ipr_cm, groups = ipr_factory  # type: ignore[misc]

           async with ipr_cm as ipr:  # type: ignore[union-attr]
               # PITFALLS §6.1: 4 MiB SO_RCVBUF. AsyncIPRoute exposes the
               # underlying socket via .asyncore.socket per Context7 verified
               # examples (RESEARCH.md A4); fall back to letting the test
               # FakeAsyncIPRoute record the setsockopt call only if real.
               sock_holder = getattr(ipr, "asyncore", None)
               if sock_holder is not None:
                   sock = getattr(sock_holder, "socket", None)
                   if sock is not None and hasattr(sock, "setsockopt"):
                       sock.setsockopt(
                           socket.SOL_SOCKET, socket.SO_RCVBUF, 4 * 1024 * 1024
                       )
               await ipr.bind(groups=groups)
               async for _msg in ipr.get():
                   event_queue.put_nowait(WakeSignal.RTNETLINK)
       ```

       Note: the pyroute2 imports are deferred (inside the function
       body, not at module level) so Windows dev hosts can import the
       module without Linux dependencies. Tests inject the
       `ipr_factory` tuple.

       ```python
       from typing import Self


       class _FakeSocket:
           def __init__(self) -> None:
               self.setsockopt_calls: list[tuple[int, int, int]] = []

       class _FakeAsyncoreHolder:
           def __init__(self) -> None:

       class FakeAsyncIPRoute:
           """Async context manager + async iterator that yields injected msgs."""

           def __init__(self) -> None:
               self.asyncore = _FakeAsyncoreHolder()
               self._queue: deque[object | OSError] = deque()
               self.bind_calls: list[int] = []
               self._closed: bool = False

           # async-context-manager surface
           async def __aenter__(self) -> Self:
               return self

           async def __aexit__(self, *exc: object) -> None:
               self._closed = True

           async def bind(self, *, groups: int) -> None:
               self.bind_calls.append(groups)

           # async-iterator surface — get() returns an async-iterable
           def get(self) -> AsyncIterator[object]:
               return _FakeMsgIter(self)

           # test mutators
           def inject_message(self, msg: object) -> None:
               self._queue.append(msg)

           def inject_enobufs(self) -> None:
               import errno
               self._queue.append(OSError(errno.ENOBUFS, "ENOBUFS"))

           def close(self) -> None:
               self._closed = True


       class _FakeMsgIter:
           def __init__(self, parent: FakeAsyncIPRoute) -> None:
               self._parent = parent

           def __aiter__(self) -> Self:
               return self

           async def __anext__(self) -> object:
               import asyncio
               while not self._parent._queue:
                   if self._parent._closed:
                       raise StopAsyncIteration
                   await asyncio.sleep(0)
               item = self._parent._queue.popleft()
               if isinstance(item, OSError):
                   raise item
               return item
       ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Kernel netlink → producer | rtnetlink socket is kernel-managed; messages are opaque to the producer (E-02 — only the wake signal is forwarded). |
| ENOBUFS → supervisor | Producer lets OSError(ENOBUFS) escape; supervisor.restart_on_crash catches Exception, backs off, restarts. The producer is intentionally "fail-fast" so the buffer can be re-allocated cleanly (close+reopen — PITFALLS §6.1). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-03-01 | DoS | ENOBUFS event flood from rtnetlink during USB hub PSU droop (PITFALLS §6.1) | mitigate | Tight read loop body (no parsing, no logging, no state) keeps consumer-side latency near-zero; 4 MiB SO_RCVBUF (16x kernel default) absorbs the typical 16-event-in-2s storm; OSError escapes to supervisor for socket close+reopen. |
| T-03-03-02 | Information Disclosure | rtnetlink message payload (link MAC addresses, link names) leaked through producer | mitigate | Producer body is `put_nowait(WakeSignal.RTNETLINK)` ONLY — payload never enters daemon state or events.jsonl. State derives from cycle re-observation which uses sysfs, not netlink payloads. |
| T-03-03-03 | Tampering | Forged netlink message via CAP_NET_ADMIN-equipped local user | accept | Daemon is already root + CAP_NET_ADMIN (NFR-30); no other suid binary. Forging requires root-equivalent access. |
| T-03-03-04 | Resource Exhaustion | pyroute2 socket leak on ungraceful exit (PITFALLS §6.3) | mitigate | `async with AsyncIPRoute() as ipr:` async context manager guarantees socket close on any exit path including CancelledError; verified by `test_aexit_called_on_cancel`. |
</threat_model>
