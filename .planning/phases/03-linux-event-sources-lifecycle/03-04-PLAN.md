          ```python
          def reopen(self) -> None:
              """Close fd, reopen at the same path, flush buffered writes.

              Called by the asyncinotify dispatcher (Plan 03-04
              event_logger/inotify_reopener.py) when logrotate moves
              the inode out from under us. R-03: window is microseconds
              in the happy path (single coroutine, no awaits between
              detect and reopen); the buffer is defense for the
              pathological case of disk-full / EPERM on the new fd.
              """
              self._reopening = True
              try:
                  old_fd = self._fd
                  self._fd = None
                  if old_fd is not None:
                      with contextlib.suppress(OSError):
                          os.close(old_fd)
                  # Re-open at the same path; logrotate `create 0640 root adm`
                  # has already created it (R-02).
                  self._fd = os.open(
                      str(self._path),
                      os.O_WRONLY | os.O_CREAT | os.O_APPEND,
                      _MODE,
                  )
                  # Flush buffered writes in FIFO order.
                  while self._reopen_buffer:
                      buffered = self._reopen_buffer.popleft()
                      os.write(self._fd, buffered)
              finally:
                  self._reopening = False
          ```

       ```python
       """ZaoLogInotifyTailer — inotify-driven Zao log reader (R-04).

       Replaces parser.py at runtime; parser.py stays as a test/replay
       helper. Satisfies the existing ZaoLogTailer Protocol so
       observer/ doesn't change.

       Handles BOTH `create` mode (MOVE_SELF/DELETE_SELF + parent-dir
       CREATE/MOVED_TO) AND `copytruncate` mode (st_size truncation
       check + opportunistic inode compare) per FR-43.1 / R-04.
       PITFALLS §8.1 prescription verbatim.
       """
       from __future__ import annotations

       import logging
       import os
       from pathlib import Path
       from typing import Protocol

       from spark_modem.zao_log.parser import ZaoLogParser
       from spark_modem.zao_log.snapshot import ZaoSnapshot

       logger = logging.getLogger(__name__)


       class _EventQueueProto(Protocol):
           def put_nowait(self, item: object) -> None: ...


       class ZaoLogInotifyTailer:
           """ZaoLogTailer Protocol satisfier; inotify-driven.

           The asyncinotify producer (asyncinotify_producer.py) calls
           on_inotify_event() for each event affecting our log path or
           parent dir. is_line_active() / snapshot() are the
           Protocol surface the observer reads.
           """

           def __init__(self, *, log_path: Path) -> None:
               self._log_path = log_path
               self._parser = ZaoLogParser(log_path)  # reuse regex/block logic
               self._last_offset: int = 0
               self._last_inode: int | None = None
               self._snapshot: ZaoSnapshot = ZaoSnapshot.unknown(
                   reason="zao_log_missing"
               )
               # Initialize from current state if file exists.
               try:
                   st = os.stat(log_path)
                   self._last_inode = st.st_ino
                   self._last_offset = 0  # cold start re-reads from beginning
                   self._snapshot = self._parser.snapshot()
               except FileNotFoundError:
                   pass  # parent-dir watch will fire IN_CREATE later

           def is_line_active(self, line_idx: int) -> bool:
               return self._snapshot.is_line_active(line_idx)

           def snapshot(self) -> ZaoSnapshot:
               return self._snapshot

           async def on_inotify_event(self, *, mask_modify: bool,
                                       mask_move_or_delete_self: bool,
                                       mask_create_or_moved_to: bool,
                                       event_path_basename: str | None,
                                       event_queue: _EventQueueProto) -> None:
               """Handle one inotify event from the asyncinotify producer.

               Caller has already extracted three orthogonal mask
               booleans + the event basename so this method's
               implementation is decoupled from asyncinotify.Mask
               specifics (Windows tests use FakeMask; production uses
               real Mask — same booleans).
               """
               if mask_move_or_delete_self:
                   # File renamed (create-mode rotation) or deleted.
                   # Reset offset; wait for parent-dir IN_CREATE.
                   self._last_offset = 0
                   self._snapshot = ZaoSnapshot.unknown(
                       reason="zao_log_missing"
                   )
                   event_queue.put_nowait(_zao_wake_signal())
                   return
               if mask_create_or_moved_to and event_path_basename == self._log_path.name:
                   # File reappeared after create-mode rotation.
                   try:
                       st = os.stat(self._log_path)
                       self._last_inode = st.st_ino
                       self._last_offset = 0
                       self._snapshot = self._parser.snapshot()
                   except FileNotFoundError:
                       return
                   event_queue.put_nowait(_zao_wake_signal())
                   return
               if mask_modify:
                   # Two interleaved cases:
                   #   1. Normal append (st_size grew)
                   #   2. copytruncate (st_size SHRANK below last_offset)
                   try:
                       st = os.stat(self._log_path)
                   except FileNotFoundError:
                       return
                   if st.st_size < self._last_offset:
                       # COPYTRUNCATE: reset offset, full re-read.
                       self._last_offset = 0
                       logger.info("zao_log copytruncate detected path=%s", self._log_path)
                   if self._last_inode is not None and st.st_ino != self._last_inode:
                       # Opportunistic inode change.
                       self._last_inode = st.st_ino
                       self._last_offset = 0
                       logger.info("zao_log inode changed path=%s", self._log_path)
                   # Re-parse — Phase 2's ZaoLogParser walks the file
                   # backwards and extracts the latest RASCOW_STAT block,
                   # so we don't need to track byte offsets precisely;
                   # snapshot() is idempotent against the current file.
                   self._snapshot = self._parser.snapshot()
                   self._last_offset = st.st_size
                   event_queue.put_nowait(_zao_wake_signal())


       def _zao_wake_signal() -> object:
           # Deferred import to avoid circular import event_sources <-> zao_log.
           from spark_modem.event_sources.supervisor import WakeSignal
           return WakeSignal.ZAO_LOG
       ```

       ```python
           rotate via IN_CREATE / IN_MOVED_TO on basename)
         - /var/log/zao/ (parent dir + the Zao log file; dual-mode
           per PITFALLS §8.1)

       Dispatches each event to the appropriate consumer based on
       which add_watch handle the event came from
       (event.watch comparison).

       PITFALLS §8.4: asyncinotify async context manager guarantees
       inotify FDs are released on shutdown; supervisor backoff caps
       FD-exhaustion rate at 60s.

       import logging
       from pathlib import Path
       from typing import Protocol

       from spark_modem.event_sources.supervisor import WakeSignal

       logger = logging.getLogger(__name__)


       class _EventQueueProto(Protocol):
           def put_nowait(self, item: object) -> None: ...


       class _EventLogReopenerProto(Protocol):
           async def on_rotate(self) -> None: ...


       class _ZaoTailerProto(Protocol):
           async def on_inotify_event(self, *, mask_modify: bool,
                                       mask_move_or_delete_self: bool,
                                       mask_create_or_moved_to: bool,
                                       event_path_basename: str | None,
                                       event_queue: _EventQueueProto) -> None: ...

           *,
           event_queue: _EventQueueProto,
           events_jsonl_path: Path,
           zao_log_path: Path,
           events_log_reopener: _EventLogReopenerProto,
           zao_tailer: _ZaoTailerProto,
           inotify_factory: object | None = None,
       ) -> None:
           """Single supervised producer watching two log directories.

           inotify_factory is an injection seam for tests
           (FakeAsyncinotify); production uses the real
           asyncinotify.Inotify().
           """
           # Defer asyncinotify import for Windows dev host friendliness.
           if inotify_factory is None:
               from asyncinotify import Inotify, Mask  # type: ignore[import-untyped]
               inotify = Inotify()
               # Real Mask values
               mask_file = (Mask.MODIFY | Mask.MOVE_SELF
                            | Mask.DELETE_SELF | Mask.CLOSE_WRITE)
               mask_parent = Mask.CREATE | Mask.MOVED_TO
           else:
               # Tests inject (FakeAsyncinotify_instance, FakeMask_class)
               inotify, fake_mask_cls = inotify_factory  # type: ignore[misc]
               mask_file = (fake_mask_cls.MODIFY | fake_mask_cls.MOVE_SELF
                            | fake_mask_cls.DELETE_SELF | fake_mask_cls.CLOSE_WRITE)
               mask_parent = fake_mask_cls.CREATE | fake_mask_cls.MOVED_TO

           # Watch handles (production code stores these and dispatches).
           events_parent_watch: object | None = None
           zao_parent_watch: object | None = None
           zao_file_watch: object | None = None

           async with inotify as ino:  # type: ignore[union-attr]
               # Watch parent dirs first (handle file-absent-at-startup;
               # PITFALLS §8.2).
               events_parent_watch = ino.add_watch(
                   events_jsonl_path.parent, mask_parent
               )
               zao_parent_watch = ino.add_watch(
                   zao_log_path.parent, mask_parent
               )
               # Watch the Zao log file directly if it exists; the
               # events.jsonl writer does NOT need a file-watch — its
               # consumer is reopen-on-rotate (R-01) and we get all
               # rotation signals from the parent-dir watch.
               if zao_log_path.exists():
                   zao_file_watch = ino.add_watch(zao_log_path, mask_file)

               async for event in ino:
                   path = getattr(event, "path", None)
                   mask = event.mask
                   # Decompose the mask into the three orthogonal
                   # booleans the consumers care about.
                   m_modify = bool(mask & mask_file & mask_file.__class__.MODIFY)
                   m_move_or_delete = bool(
                       mask & (mask_file.__class__.MOVE_SELF
                                | mask_file.__class__.DELETE_SELF)
                   )
                   m_create_or_moved_to = bool(
                       mask & (mask_parent.__class__.CREATE
                                | mask_parent.__class__.MOVED_TO)
                   )
                   basename = path.name if path is not None else None

                   # Dispatch by watch handle (R-01: one producer, two consumers).
                   if event.watch == events_parent_watch:
                       # events.jsonl rotation only — the writer doesn't
                       # need MODIFY signals (it's the writer; it knows
                       # what it wrote).
                       if m_create_or_moved_to and basename == events_jsonl_path.name:
                           await events_log_reopener.on_rotate()
                           event_queue.put_nowait(WakeSignal.EVENTS_LOG_ROTATED)
                       continue
                   if event.watch in (zao_parent_watch, zao_file_watch):
                       # Re-watch on parent CREATE if needed.
                       if (event.watch == zao_parent_watch
                               and m_create_or_moved_to
                               and basename == zao_log_path.name
                               and zao_file_watch is None):
                           zao_file_watch = ino.add_watch(zao_log_path, mask_file)
                       await zao_tailer.on_inotify_event(
                           mask_modify=m_modify,
                           mask_move_or_delete_self=m_move_or_delete,
                           mask_create_or_moved_to=m_create_or_moved_to,
                           event_path_basename=basename,
                           event_queue=event_queue,
                       )
                       continue
       ```

| logrotate (external) → events.jsonl writer | Logrotate runs as root via cron and recreates the file with `create 0640 root adm`. Daemon writer reopens on rotation; trust the file path is daemon-owned because it lives in `ReadWritePaths=` of the systemd unit. |
