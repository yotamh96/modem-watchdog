       ```python
       """udev producer — push WakeSignal.UDEV on Sierra-VID USB events.

       PITFALLS §7.1 PRESCRIPTIVE: NEVER pyudev.MonitorObserver — the
       observer thread crashes silently under bulk events (pyudev #194,
       #363, #402). Single-threaded loop.add_reader is the only path.

       PITFALLS §7.2: react on `bind` (driver attached) for cdc-wdm —
       sysfs not fully populated on `add`. We forward `add`, `remove`,
       `bind`, `unbind` as wake signals; cycle re-observation reads
       sysfs, retries on transient EAGAIN/ENOENT.

       E-02 (CONTEXT.md): producer pushes opaque WakeSignal only.
       State derives from cycle re-observation, NEVER from event
       payload.
       """
       from __future__ import annotations

       import asyncio
       import logging
       from typing import Final, Protocol

       from spark_modem.event_sources.supervisor import WakeSignal

       logger = logging.getLogger(__name__)

       _SIERRA_VID: Final[str] = "1199"


       class _EventQueueProto(Protocol):
           def put_nowait(self, item: object) -> None: ...


       class _MonitorProto(Protocol):
           """Subset of pyudev.Monitor + Device the producer touches.

           Co-located so test code injects FakeUdevMonitor without
           monkeypatching pyudev itself (PITFALLS §14.1 spirit).
           """
           def filter_by(self, **kwargs: str) -> None: ...
           def set_receive_buffer_size(self, n_bytes: int) -> None: ...
           def start(self) -> None: ...
           def fileno(self) -> int: ...
           def poll(self, timeout: float = 0) -> object: ...


       def _build_default_monitor() -> _MonitorProto:
           """Construct a real pyudev.Monitor. Linux-only; tests inject."""
           import pyudev  # type: ignore[import-untyped]
           context = pyudev.Context()
           monitor = pyudev.Monitor.from_netlink(context)
           monitor.filter_by(subsystem="usb")
           monitor.set_receive_buffer_size(4 * 1024 * 1024)  # PITFALLS §7.3
           monitor.start()
           return monitor


       async def run_udev_producer(
           *,
           event_queue: _EventQueueProto,
           sierra_vid: str = _SIERRA_VID,
           monitor: _MonitorProto | None = None,
       ) -> None:
           """Subscribe to USB add/remove/bind/unbind for Sierra-VID modems.

           Pushes WakeSignal.UDEV on every matching event. NEVER reads
           sysfs (E-02 — state derives from cycle re-observation).
           """
           loop = asyncio.get_running_loop()
           mon = monitor if monitor is not None else _build_default_monitor()

           def on_readable() -> None:
               # Drain the kernel queue in a tight loop; PITFALLS §7.1.
               while True:
                   device = mon.poll(timeout=0)
                   if device is None:
                       break
                   action = getattr(device, "action", None)
                   vid = device.get("ID_VENDOR_ID") if hasattr(device, "get") else None
                   if action in ("add", "remove", "bind", "unbind") and vid == sierra_vid:
                       event_queue.put_nowait(WakeSignal.UDEV)

           loop.add_reader(mon.fileno(), on_readable)
           try:
               # Sleep forever; supervisor cancels via CancelledError.
               await asyncio.Future()
           finally:
               loop.remove_reader(mon.fileno())
       ```

       Note: the `import pyudev` lives inside `_build_default_monitor()`
       so the module imports cleanly on Windows dev hosts (pyudev is
       Linux-only via libudev.so.1). Tests inject the monitor and never
       trigger the import.

    A. WRITE `src/spark_modem/inventory/netns.py`:

       ```python
       """netns derivation for a USB device's netns assignment (E-05).

       Open Question 4 (RESEARCH.md): three options exist for reading
       the netns of a cdc-wdm worker. We pick option-(a) — read the
       symlink `<usb_dev_path>/<intf>/net/<wwanN>/device/ns/net` if
       present; fall back to None on FileNotFoundError. The bench
       Jetson is single-namespace (ns=None); production fleet may
       wrap modems in per-line netns.

       PITFALLS §6.2 reminder: this function ONLY READS sysfs; the
       daemon NEVER calls setns() from the asyncio loop. Plan 03-02
       Task 2 wires QmiWrapper to prepend `ip netns exec <ns>` —
       subprocess does its own setns in a forked child.
       """
       from __future__ import annotations

       from pathlib import Path


       def derive_ns(usb_dev_path: Path) -> str | None:
           """Return the netns name for the cdc-wdm under usb_dev_path, or None.

           Looks for `<usb_dev_path>/.../net/wwan*/device/ns/net` (sysfs
           symlink); reads the readlink target's basename. On any IO
           error returns None — the bench Jetson is single-namespace so
           None is the expected default.
           """
           # Find the wwanN interface (mirrors _find_wwan_iface in sysfs.py).
           for net in usb_dev_path.rglob("net/wwan*"):
               if not (net.is_dir() or net.is_symlink()):
                   continue
               ns_link = net / "device" / "ns" / "net"
               try:
                   target = ns_link.readlink()
               except (FileNotFoundError, OSError, NotADirectoryError):
                   return None
               # readlink returns e.g. PosixPath("net:[4026531992]") —
               # strip "net:[" and "]" to get the inode; the netns NAME
               # (what `ip netns add foo` produces) is in
               # /var/run/netns/<name> with the same inode. We don't
               # need the name on the bench Jetson (None is correct);
               # if a future deployment uses named netns, walk
               # /var/run/netns/ matching the inode. Phase 3 returns
               # None when the link exists but doesn't resolve to a
               # named netns.
               return _resolve_netns_name(target)
           return None


       def _resolve_netns_name(ns_link_target: Path) -> str | None:
           """Match a `net:[<inode>]` link target to a name in /var/run/netns/."""
           name = str(ns_link_target)
           if not name.startswith("net:["):
               return None
           inode_str = name[5:].rstrip("]")
           try:
       ```
