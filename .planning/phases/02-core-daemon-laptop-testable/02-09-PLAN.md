    """CLI helpers — FixtureRunner that loads canned qmicli output from
    tests/fixtures/qmicli/<intent>/<libqmi-version>/<scenario>.txt or any
    user-supplied directory with the same shape.
    """
    from __future__ import annotations

    from pathlib import Path

    from spark_modem.qmi.parsers._header import strip_header
    from spark_modem.subproc.result import CompletedProcess


    class FixtureRunner:
        """Reads canned qmicli stdout from disk based on argv intent.

        Resolution: argv contains a flag like '--nas-get-signal-info'; we
        find the file at <fixture_dir>/get_signal/<libqmi-version>/<scenario>.txt
        (using the FIRST file found in the highest-priority libqmi version
        directory; default-version=1.30; scenario defaults to 'lte_strong').
        """

        _INTENT_MAP: dict[str, str] = {
            "--nas-get-signal-info": "get_signal",
            "--nas-get-serving-system": "get_serving_system",
            "--uim-get-card-status": "get_sim_state",
            "--wds-get-packet-service-status": "get_data_session",
            "--wds-get-current-settings": "get_current_settings",
            "--dms-get-operating-mode": "get_operating_mode",
        }

        def __init__(
            self,
            *,
            fixture_dir: Path,
            libqmi_version: str = "1.30",
            scenario: str = "lte_strong",
        ) -> None:
            self._dir = Path(fixture_dir)
            self._version = libqmi_version
            self._scenario = scenario

        async def run(
            self, argv: list[str], *, timeout_s: float,
            stdin: bytes | None = None, env: dict[str, str] | None = None,
        ) -> CompletedProcess:
            del timeout_s, stdin, env
            intent = self._intent_of(argv)
            if intent is None:
                return CompletedProcess.make(
                    argv=argv, exit_code=2, stdout=b"",
                    stderr=b"FixtureRunner: unknown qmicli intent\n",
                    duration_monotonic=0.0,
                )
            scenario_path = self._scenario_path(intent)
            if scenario_path is None:
                return CompletedProcess.make(
                    argv=argv, exit_code=2, stdout=b"",
                    stderr=f"FixtureRunner: no fixture for intent={intent}\n".encode(),
                    duration_monotonic=0.0,
                )
            stdout = strip_header(scenario_path.read_bytes())
            return CompletedProcess.make(
                argv=argv, exit_code=0, stdout=stdout, stderr=b"",
                duration_monotonic=0.001,
            )

        def _intent_of(self, argv: list[str]) -> str | None:
            for elem in argv:
                # Trailing-= flags like --wds-get-profile-settings=3gpp,1
                key = elem.split("=", 1)[0]
                if key in self._INTENT_MAP:
                    return self._INTENT_MAP[key]
                if key == "--wds-get-profile-settings":
                    return "get_profile_settings"
            return None

        def _scenario_path(self, intent: str) -> Path | None:
            base = self._dir / intent / self._version
            if not base.is_dir():
                return None
            specific = base / f"{self._scenario}.txt"
            if specific.is_file():
                return specific
            # Fall back to any fixture file in the directory
            for entry in sorted(base.iterdir()):
                if entry.is_file() and entry.suffix == ".txt":
                    return entry
            return None
    ```

        # status
        p = sub.add_parser("status", help="Print status.json contents")
        p.add_argument("--state-root", type=str, default=None)
        p.set_defaults(func=status_cmd.run)

        # ctl
        ctl_parser = sub.add_parser("ctl", help="Operator control commands")
        ctl_sub = ctl_parser.add_subparsers(dest="ctl_cmd", required=True)

        # ctl history
        ph = ctl_sub.add_parser("history", help="Per-modem event history")
        ph.add_argument("--modem", type=str, default=None,
                        help="usb_path or cdc-wdmN")
        ph.add_argument("--since", type=str, default=None,
                        help="Duration (e.g. '1h', '30m')")
        ph.set_defaults(func=ctl_history.run)

        # ctl maintenance
        pm = ctl_sub.add_parser("maintenance",
                                help="Maintenance window control")
        pm_sub = pm.add_subparsers(dest="maint_cmd", required=True)
        pm_on = pm_sub.add_parser("on", help="Enable maintenance window")
        pm_on.add_argument("--duration", type=str, required=True,
                           help="Mandatory; e.g. '2h', max 8h")
        pm_on.set_defaults(func=ctl_maintenance.run_on)
        pm_off = pm_sub.add_parser("off", help="Disable maintenance window")
        pm_off.set_defaults(func=ctl_maintenance.run_off)
        pm_st = pm_sub.add_parser("status", help="Show maintenance status")

        # ctl support-bundle
        psb = ctl_sub.add_parser("support-bundle",
                                 help="Build redacted support tarball")

                         help="Output path (default: /var/lib/.../support-bundles/...)")
        psb.set_defaults(func=ctl_support_bundle.run)

        return parser


    def main(argv: list[str] | None = None) -> int:
        parser = _build_parser()
        args = parser.parse_args(argv)
        return asyncio.run(args.func(args))


    if __name__ == "__main__":
        sys.exit(main())

    """PII redaction (C-04 — one-way, consistent across the bundle)."""
    from __future__ import annotations

    import hashlib
    import re
    from typing import Any


    def redact_pii(value: str) -> str:
        """Returns '<redacted:<sha256[:8]>>' — one-way, consistent.

        Same input → same redacted form across the support bundle, so
        identity-correlation is preserved without exporting PII.
        """
        digest = hashlib.sha256(value.encode("utf-8")).hexdigest()[:8]
        return f"<redacted:{digest}>"


    _PII_FIELDS: frozenset[str] = frozenset({"iccid", "imsi"})

    _SHA8_PATTERN = re.compile(r"<redacted:[0-9a-f]{8}>")


    def redact_iccid_imsi_in_dict(obj: Any) -> Any:        # noqa: ANN401
        """Recursively redact ICCID/IMSI fields in a parsed JSON structure."""
        if isinstance(obj, dict):
            out: dict[str, Any] = {}
            for k, v in obj.items():
                if k.lower() in _PII_FIELDS and isinstance(v, str):
                    out[k] = redact_pii(v)
                else:
                    out[k] = redact_iccid_imsi_in_dict(v)
            return out
        if isinstance(obj, list):
            return [redact_iccid_imsi_in_dict(item) for item in obj]
        return obj


    def redact_webhook_url_to_host_only(url: str) -> str:
        """Strip path/query from URL; keep scheme + netloc only."""
        from urllib.parse import urlsplit
        parts = urlsplit(url)
        if not parts.netloc:
            return url
        return f"{parts.scheme}://{parts.netloc}/"

    """ctl history --modem= --since= — read events.jsonl + rotated siblings.

    FR-50.1: filter by modem (usb_path canonical or cdc-wdmN alias).
    Phase 3 logrotate creates `events.jsonl.1`, `events.jsonl.2.gz`, etc.;
    Phase 2 reader supports both compressed-gzip and plain text rotated
    siblings, but only opens them if they exist.
    """
    from __future__ import annotations

    import argparse
    import gzip
    import json
    import re
    import sys
    from collections.abc import Iterable
    from datetime import UTC, datetime, timedelta
    from pathlib import Path

    from spark_modem.wire.events import Event, EventAdapter

    _DURATION_RE = re.compile(r"^(\d+)([hms])$")


    def parse_since(s: str) -> float:
        m = _DURATION_RE.match(s.strip().lower())
        if not m:
            raise ValueError(f"invalid --since: {s!r}")
        n = int(m.group(1))
        unit = m.group(2)
        if unit == "h":
            return float(n * 3600)
        if unit == "m":
            return float(n * 60)
        return float(n)


    def read_events_with_rotated_siblings(events_log_path: Path) -> Iterable[Event]:
        """Yield events from events.jsonl + events.jsonl.1, .2, ... + .gz variants."""
        candidates: list[Path] = []
        # Plain rotated: events.jsonl, events.jsonl.1, events.jsonl.2, ...
        candidates.append(events_log_path)
        for i in range(1, 10):
            sib = events_log_path.with_suffix(f".jsonl.{i}")
            if sib.is_file():
                candidates.append(sib)
            sibgz = events_log_path.with_suffix(f".jsonl.{i}.gz")
            if sibgz.is_file():
                candidates.append(sibgz)
        for path in reversed(candidates):  # oldest first so output is chronological
            if not path.is_file():
                continue
            opener = gzip.open if path.suffix == ".gz" else open
            with opener(path, "rb") as fh:
                for raw_line in fh:
                    line = raw_line.strip()
                    if not line:
                        continue
                    try:
                        yield EventAdapter.validate_json(line)
                    except Exception:  # noqa: BLE001 — corrupt lines skipped, not raised
                        continue


    def filter_events(
        events: Iterable[Event],
        *,
        modem: str | None,
        since_seconds: float | None,
    ) -> list[Event]:
        cutoff_iso: str | None = None
        if since_seconds is not None:
            cutoff_iso = (datetime.now(UTC) - timedelta(seconds=since_seconds)).isoformat()
        out: list[Event] = []
        for event in events:
            if cutoff_iso is not None and event.ts_iso < cutoff_iso:
                continue
            if modem is not None:
                # Each event variant has a different field for the modem id;
                # match on usb_path attr if present.
                ep = getattr(event, "usb_path", None) or getattr(event, "modem_usb_path", None)
                if ep is None or (modem != ep and not modem.startswith("cdc-wdm")):
                    continue
                # cdc-wdmN alias matching is best-effort — the daemon's identity
                # map ties usb_path ↔ cdc_wdm; without it, only exact usb_path matches.
            out.append(event)
        return out


    async def run(args: argparse.Namespace) -> int:
        events_log = Path("/var/log/spark-modem-watchdog/events.jsonl")
        since_seconds: float | None = None
        if args.since is not None:
            try:
                since_seconds = parse_since(args.since)
            except ValueError as exc:
                print(f"ctl history: {exc}", file=sys.stderr)
                return 2
        events = list(read_events_with_rotated_siblings(events_log))
        events = filter_events(events, modem=args.modem, since_seconds=since_seconds)
        for ev in events:
            print(json.dumps(ev.model_dump(mode="json")))
        return 0

        with tarfile.open(target, "w:gz") as tar:
            # Last 200 events (redacted)
            await _add_events(tar, events_log)
            # state/by-usb/*.json (redacted)
            await _add_state_files(tar, sr)
            # globals.json (raw — no PII)
            await _add_globals(tar, store)
            # status.json (raw)
            await _add_status(tar, sr)
            # carriers.yaml + conf.d (excluding hmac-secret)
            _add_conf_d(tar)
            # webhook-url metadata (host only)
            _add_metadata(tar, webhook_url=webhook_url_for_redaction)

        target.chmod(_BUNDLE_MODE)
        return target


    async def _add_events(tar: tarfile.TarFile, events_log: Path) -> None:
        events = list(read_events_with_rotated_siblings(events_log))[-_LAST_N_EVENTS:]
        body_parts: list[str] = []
        for ev in events:
            d = ev.model_dump(mode="json")
            d = redact_iccid_imsi_in_dict(d)
            body_parts.append(json.dumps(d))
        body = ("\n".join(body_parts) + "\n").encode("utf-8")
        info = tarfile.TarInfo(name="events.last200.jsonl")
        info.size = len(body)
        info.mode = 0o600
        tar.addfile(info, io.BytesIO(body))


    async def _add_state_files(tar: tarfile.TarFile, state_root: Path) -> None:
        by_usb = state_root / "state" / "by-usb"
        if not by_usb.is_dir():
            return
        for f in sorted(by_usb.iterdir()):
            if f.suffix == ".json" and ".from-v" not in f.name:
                raw = json.loads(f.read_bytes())
                redacted = redact_iccid_imsi_in_dict(raw)
                body = (json.dumps(redacted, indent=2) + "\n").encode("utf-8")
                info = tarfile.TarInfo(name=f"state/by-usb/{f.name}")
                info.size = len(body)
                info.mode = 0o600
                tar.addfile(info, io.BytesIO(body))


    async def _add_globals(tar: tarfile.TarFile, store: StateStore) -> None:
        load = await store.load_globals()
        body = (load.state.model_dump_json(by_alias=True, indent=2) + "\n").encode("utf-8")
        info = tarfile.TarInfo(name="globals.json")
        info.size = len(body)
        info.mode = 0o600
        tar.addfile(info, io.BytesIO(body))


    async def _add_status(tar: tarfile.TarFile, state_root: Path) -> None:
        path = state_root / "status.json"
        if not path.is_file():
            return
        body = path.read_bytes()
        info = tarfile.TarInfo(name="status.json")
        info.size = len(body)
        info.mode = 0o600
        tar.addfile(info, io.BytesIO(body))


    def _add_conf_d(tar: tarfile.TarFile) -> None:
        conf_d = Path("/etc/spark-modem-watchdog/conf.d")
        if not conf_d.is_dir():
            return
        for entry in sorted(conf_d.iterdir()):
            if entry.is_file() and entry.name != "hmac-secret":
                body = entry.read_bytes()
                info = tarfile.TarInfo(name=f"conf.d/{entry.name}")
                info.size = len(body)
                info.mode = 0o600
                tar.addfile(info, io.BytesIO(body))


    def _add_metadata(tar: tarfile.TarFile, *, webhook_url: str | None) -> None:
        meta = {
            "tool_version": "spark-modem-watchdog/2.0.0",
            "host": socket.gethostname(),
            "ts_iso": datetime.now(UTC).isoformat(),
            "webhook_url": redact_webhook_url_to_host_only(webhook_url) if webhook_url else None,
        }
        body = (json.dumps(meta, indent=2) + "\n").encode("utf-8")
        info = tarfile.TarInfo(name="metadata.json")
        info.size = len(body)
        info.mode = 0o600
        tar.addfile(info, io.BytesIO(body))


    async def run(args: argparse.Namespace) -> int:
        out_path = Path(args.out) if args.out else None
        try:
            target = await build_support_bundle(out_path=out_path)
        except Exception as exc:
            print(f"ctl support-bundle: failed: {exc}", file=sys.stderr)
            return 1
        print(str(target.resolve()))
        return 0
    ```
