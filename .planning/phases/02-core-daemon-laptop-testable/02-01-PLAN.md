    """FakeRunner — argv→CompletedProcess map for hardware-free Phase 2 tests."""
    from __future__ import annotations

    from spark_modem.subproc.result import CompletedProcess


    class FakeRunner:
        def __init__(self) -> None:
            self._responses: dict[tuple[str, ...], CompletedProcess] = {}
            self._calls: list[list[str]] = []

        def register(self, argv: list[str], result: CompletedProcess) -> None:
            self._responses[tuple(argv)] = result

        @property
        def calls(self) -> list[list[str]]:
            return list(self._calls)

        async def run(
            self,
            argv: list[str],
            *,
            timeout_s: float,
            stdin: bytes | None = None,
            env: dict[str, str] | None = None,
        ) -> CompletedProcess:
            del timeout_s, stdin, env  # signature parity only
            self._calls.append(list(argv))
            key = tuple(argv)
            if key not in self._responses:
                raise KeyError(f"FakeRunner: no canned response for {argv!r}")
            return self._responses[key]
    ```

    """FakeClock — instance-method clock for deterministic asyncio tests."""
    from __future__ import annotations

    from datetime import UTC, datetime, timedelta, tzinfo


    class FakeClock:
        def __init__(
            self,
            *,
            start_monotonic: float = 0.0,
            start_wall: datetime | None = None,
        ) -> None:
            self._monotonic = start_monotonic
            self._wall = start_wall or datetime(2026, 1, 1, tzinfo=UTC)

        def monotonic(self) -> float:
            return self._monotonic

        def elapsed_since(self, t0_monotonic: float) -> float:
            return max(0.0, self._monotonic - t0_monotonic)

        def wall_clock_iso(self, *, tz: tzinfo | None = None) -> str:
            target = self._wall if tz is None else self._wall.astimezone(tz)
            return target.isoformat()

        def advance(self, seconds: float) -> None:
            if seconds < 0:
                raise ValueError(f"FakeClock.advance: seconds must be >= 0; got {seconds}")
            self._monotonic += seconds
            self._wall = self._wall + timedelta(seconds=seconds)
    ```
