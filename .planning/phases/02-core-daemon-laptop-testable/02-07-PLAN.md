    ```python
    """status.json writer — atomic, every cycle (FR-41, O-01)."""
    from __future__ import annotations

    from pathlib import Path

    from spark_modem.state_store.atomic import atomic_write_bytes
    from spark_modem.wire.status import StatusReport


    def write_status_json(path: Path | str, report: StatusReport) -> None:
        """Atomic write: temp + rename + dir-fsync (delegates to Phase 1 atomic).

        Caller invokes once per cycle, after policy + actions, before sleep.
        Per O-01: marginal cost vs the per-modem state writes that happen
        every cycle anyway.
        """
        target = Path(path)
        payload = report.model_dump_json(by_alias=True).encode("utf-8")
        atomic_write_bytes(target, payload)
    ```

    ```python
    """Typed metric registry — single chokepoint for every Prom metric we set/inc.

    Cardinality discipline (ADR-0013):
      - modem_state_value{modem} is a SINGLE Gauge whose VALUE is the integer
        state code (0..4 stable mapping per state_to_int). NOT one-hot.
      - state_duration_seconds{modem, state} histogram has a fixed
        bucket array [1, 5, 15, 60, 300, 1800, 7200, 86400] (O-02).
      - webhook_delivery_total{result} has a closed enum of result values
        (sent / failed / dropped / coalesced / skipped_no_url / skipped_no_dns).
        See O-04 / FR-44 / NFR-21.1.

    Phase 2 supports a Windows-friendly mode: when `prometheus_client` is
    installed but the AF_UNIX server cannot start (Win10 < 1803 or wsgiref
    quirks), the registry still records — there's just no scrape endpoint.
    Tests assert metric updates without scraping by reading the registry's
    internal collectors via `prometheus_client.REGISTRY.collect()`.
    """
    from __future__ import annotations

    from prometheus_client import Counter, Gauge, Histogram

    from spark_modem.wire.enums import ActionKind, ActionResult

    _CYCLE_BUCKETS = (0.5, 1, 2, 4, 8, 16, 32)
    _STATE_DURATION_BUCKETS = (1, 5, 15, 60, 300, 1800, 7200, 86400)


    class MetricRegistry:
        """Single-instance per process. Cycle driver constructs once at startup.

        IMPORTANT: prometheus_client metrics are global (REGISTRY singleton);
        constructing MetricRegistry twice in the same Python process raises
        ValueError on duplicate metric registration.  Tests that need a
        fresh registry call `prometheus_client.REGISTRY.unregister(coll)`
        in a fixture teardown — see test_metrics_registry.py.
        """

        def __init__(self) -> None:
            self._actions_total = Counter(
                "actions_total",
                "Recovery actions executed by kind, modem, and result.",
                ["kind", "modem", "result"],
            )
            self._signal_rsrp_dbm = Gauge(
                "signal_rsrp_dbm", "RSRP in dBm per modem.", ["modem"],
            )
            self._signal_rsrq_db = Gauge(
                "signal_rsrq_db", "RSRQ in dB per modem.", ["modem"],
            )
            self._signal_snr_db = Gauge(
                "signal_snr_db", "SNR in dB per modem.", ["modem"],
            )
            self._cycle_duration_seconds = Histogram(
                "cycle_duration_seconds", "Cycle duration in seconds.",
                buckets=_CYCLE_BUCKETS,
            )
            self._modem_state_value = Gauge(
                "modem_state_value",
                "Per-modem state encoded as integer (ADR-0013): "
                "0=unknown, 1=healthy, 2=degraded, 3=recovering, 4=exhausted.",
                ["modem"],
            )
            self._state_duration_seconds = Histogram(
                "state_duration_seconds",
                "Time spent in each state per modem.",
                ["modem", "state"], buckets=_STATE_DURATION_BUCKETS,
            )
            self._cycle_drift_seconds = Gauge(
                "cycle_drift_seconds",
                "Signed gauge: now_monotonic - expected_next_cycle_monotonic, "
                "clamped to >=0 (O-03).",
            )
            self._webhook_delivery_total = Counter(
                "webhook_delivery_total",
                "Webhook deliveries by result (sent / failed / dropped / "
                "coalesced / skipped_no_url / skipped_no_dns).",
                ["result"],
            )
            self._daemon_self_health = Counter(
                "daemon_self_health",
                "Daemon self-health events.  kind values: rss (psutil tripwire).",
                ["kind"],
            )

        # -------- recording API --------

        def record_action(
            self, kind: ActionKind, modem: str, result: ActionResult,
        ) -> None:
            self._actions_total.labels(
                kind=kind.value, modem=modem, result=result.value,
            ).inc()

        def record_signal(
            self, modem: str, *, rsrp: int | None, rsrq: float | None, snr: float | None,
        ) -> None:
            if rsrp is not None:
                self._signal_rsrp_dbm.labels(modem=modem).set(float(rsrp))
            if rsrq is not None:
                self._signal_rsrq_db.labels(modem=modem).set(float(rsrq))
            if snr is not None:
                self._signal_snr_db.labels(modem=modem).set(float(snr))

        def observe_cycle_duration(self, seconds: float) -> None:
            self._cycle_duration_seconds.observe(max(0.0, seconds))

        def set_modem_state(self, modem: str, value: int) -> None:
            """ADR-0013 integer encoding. Caller passes state_to_int(state)."""
            self._modem_state_value.labels(modem=modem).set(value)

        def observe_state_duration(
            self, modem: str, state: str, seconds: float,
        ) -> None:
            self._state_duration_seconds.labels(modem=modem, state=state).observe(
                max(0.0, seconds)
            )

        def set_cycle_drift(self, seconds: float) -> None:
            """O-03: always >= 0 (negative drift indicates monotonic regression — clamp)."""
            self._cycle_drift_seconds.set(max(0.0, seconds))

        def record_webhook_delivery(self, result: str) -> None:
    ```

    ```python
    """Prometheus UDS exporter — make_wsgi_app() over AF_UNIX socket.

    RESEARCH §2.6 verbatim:
      - _UnixWSGIServer subclasses UnixStreamServer + WSGIServer
      - server_bind() does NOT call setsockopt(SO_REUSEADDR) — UDS
        kernels can return ENOPROTOOPT (Linux 5.10-tegra in known set).
      - Stale socket file is cleaned with unlink(missing_ok=True) before bind.
      - Socket mode is 0o660 so adm-group readers can scrape.
      - Server is run via asyncio.to_thread; serve_forever blocks the thread,
        not the asyncio loop.

    Windows: AF_UNIX requires Win10 1803+ + Python 3.9+.  Tests are
    skipped on win32; daemon production target is Linux/aarch64.
    """
    from __future__ import annotations

    import socket
    import sys
    from pathlib import Path
    from socketserver import UnixStreamServer
    from wsgiref.simple_server import WSGIRequestHandler, WSGIServer

    from prometheus_client import make_wsgi_app


    class _UnixWSGIServer(UnixStreamServer, WSGIServer):
        """WSGIServer subclass that binds an AF_UNIX socket instead of TCP."""

        address_family = socket.AF_UNIX

        def server_bind(self) -> None:
            # Do NOT call setsockopt(SO_REUSEADDR) — UDS sockets don't need
            # it; some kernels return ENOPROTOOPT.
            UnixStreamServer.server_bind(self)
            # wsgiref expects setup_environ() — sets SERVER_NAME etc.;
            # values are nonsense on UDS but the WSGI handler does not care.
            self.setup_environ()


    def start_metrics_server(socket_path: Path | str) -> _UnixWSGIServer:
        """Bind the AF_UNIX socket and return a server ready for serve_forever.

        Caller wraps the returned server in `asyncio.create_task(asyncio.to_thread(srv.serve_forever))`.
        """
        if sys.platform == "win32":
            raise RuntimeError(
                "Prometheus UDS exporter requires AF_UNIX socket support; "
                "Windows dev hosts use FakeMetricsServer in tests."
            )
        target = Path(socket_path)
        target.parent.mkdir(parents=True, exist_ok=True)
        target.unlink(missing_ok=True)   # PITFALLS §13.3
        server = _UnixWSGIServer(str(target), WSGIRequestHandler)
        server.set_app(make_wsgi_app())
        target.chmod(0o660)              # 0660 root:adm
        return server
    ```
