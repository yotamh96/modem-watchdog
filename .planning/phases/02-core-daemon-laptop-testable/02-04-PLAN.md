    `<sysfs_root>/bus/usb/devices/` for entries whose `idVendor` is `1199`

    - `usb_path` = the entry's basename (e.g. `2-3.1.1`)
    - `cdc_wdm` = found by walking the entry's children to find a `cdc-wdmN`
      under any `qmi_wwanN` interface
    - `iface` = corresponding `wwan0` net interface
    - `ns`, `line` = `None` and derived-from-usb-path-suffix respectively

    ```python
    """SysfsInventory — Phase 2 implementation of InventorySource via sysfs walk.

    Reads /sys/bus/usb/devices/ for VID:PID 1199:9091 (Sierra EM7421) entries.
    Phase 3 swaps in a UdevInventory backed by pyudev.Monitor; observer/
    does not change.
    """
    from __future__ import annotations

    from pathlib import Path

    from spark_modem.inventory.descriptor import ModemDescriptor

    _SIERRA_VID = "1199"
    _EM7421_PID = "9091"


    class SysfsInventory:
        """Walks /sys/bus/usb/devices/ for VID:PID 1199:9091 (Sierra EM7421)."""

        def __init__(self, *, sysfs_root_override: Path | None = None) -> None:
            self._sysfs_root = sysfs_root_override or Path("/sys")

        async def scan(self) -> list[ModemDescriptor]:
            usb_devices_dir = self._sysfs_root / "bus" / "usb" / "devices"
            if not usb_devices_dir.is_dir():
                return []
            descriptors: list[ModemDescriptor] = []
            for entry in sorted(usb_devices_dir.iterdir()):
                if not entry.is_dir() and not entry.is_symlink():
                    continue
                resolved = entry.resolve()
                vid_file = resolved / "idVendor"
                pid_file = resolved / "idProduct"
                if not (vid_file.exists() and pid_file.exists()):
                    continue
                vid = vid_file.read_text(encoding="ascii").strip().lower()
                pid = pid_file.read_text(encoding="ascii").strip().lower()
                if vid != _SIERRA_VID or pid != _EM7421_PID:
                    continue
                usb_path = entry.name
                line = self._line_from_usb_path(usb_path)
                cdc_wdm = self._find_cdc_wdm(resolved)
                iface = self._find_wwan_iface(resolved)
                if cdc_wdm is None:
                    continue   # not yet enumerated — caller retries next cycle
                descriptors.append(ModemDescriptor(
                    line=line,
                    cdc_wdm=cdc_wdm,
                    usb_path=usb_path,
                    ns=None,        # Phase 3 derives from netns
                    iface=iface,
                ))
            return descriptors

        @staticmethod
        def _line_from_usb_path(usb_path: str) -> int:
            """Map '2-3.1.1' -> 1 (the last dotted component is the line index).

            On the production Jetson the four modems sit at 2-3.1.1..4.
            Phase 3 may surface a more sophisticated mapping; Phase 2 uses
            the trailing component.
            """
            tail = usb_path.rsplit(".", 1)[-1]
            try:
                value = int(tail)
            except ValueError:
                return 1
            return value if 1 <= value <= 99 else 1

        @staticmethod
        def _find_cdc_wdm(resolved: Path) -> str | None:
            """Search children for a cdc-wdmN node and return the basename.

            sysfs layout (paraphrased):
              <usb_dev>/<intf>/usbmisc/cdc-wdm0/
              <usb_dev>/<intf>/net/wwan0/
            """
            for misc in resolved.rglob("usbmisc/cdc-wdm*"):
                if misc.is_dir() or misc.is_symlink():
                    return misc.name
            return None

        @staticmethod
        def _find_wwan_iface(resolved: Path) -> str | None:
            for net in resolved.rglob("net/wwan*"):
                if net.is_dir() or net.is_symlink():
                    return net.name
            return None
    ```

        def monotonic(self) -> float: ...
        def wall_clock_iso(self) -> str: ...

        qmi_factory: QmiFactory,
        zao: ZaoLogTailer,
        clock: ClockProto,
        *,
        timeout_s: float = DEFAULT_PROBE_TIMEOUT_S,
    ) -> list[ModemSnapshot]:
        """Run all per-modem probes in parallel and return one ModemSnapshot each.

        Order of returned snapshots matches the order of `modems`.
        """
        if not modems:
            return []
        async with asyncio.TaskGroup() as tg:
            tasks = [
                tg.create_task(_probe_one(m, qmi_factory, zao, clock, timeout_s))
                for m in modems
            ]
        return [t.result() for t in tasks]


    async def _probe_one(
        modem: ModemDescriptor,
        qmi_factory: QmiFactory,
        zao: ZaoLogTailer,
        clock: ClockProto,
        timeout_s: float,
    ) -> ModemSnapshot:
        """Per-modem probe with own try/except — never propagates to TaskGroup.

        Steps:
          1. Check Zao gate: if line is bonding-active under Zao, return
             zao-active snapshot with zero issues (FR-10).
          2. Else run qmicli queries through QmiWrapper under asyncio.timeout(8s).
          3. Catch TimeoutError → timed_out snapshot.
          4. Catch any other Exception → errored snapshot, log + continue (NFR-11).
        """
        if zao.is_line_active(modem.line):
            return _zao_active_snapshot(modem)
        try:
            async with asyncio.timeout(timeout_s):
                return await probe_modem_to_snapshot(modem, qmi_factory(modem), clock)
        except TimeoutError:
            logger.warning("probe timed out for %s", modem.usb_path)
            return _timed_out_snapshot(modem)
        except Exception:                       # NFR-11: never crash the cycle
            logger.exception("probe failed for %s", modem.usb_path)
            return _errored_snapshot(modem)


    def _zao_active_snapshot(modem: ModemDescriptor) -> ModemSnapshot:
        return ModemSnapshot(
            usb_path=modem.usb_path,
            cdc_wdm=modem.cdc_wdm,
            usb_speed=None,
            operating_mode=None,
            sim_state=None,
            registration=None,
            issues=[],   # FR-10: never act on a Zao-active line
        )


    def _timed_out_snapshot(modem: ModemDescriptor) -> ModemSnapshot:
        # No issues recorded — policy treats absent observation as 'unknown'
        # transition (RECOVERY_SPEC §3.1: unknown is action-gated like healthy).
        return ModemSnapshot(
            usb_path=modem.usb_path,
            cdc_wdm=modem.cdc_wdm,
        )


    def _errored_snapshot(modem: ModemDescriptor) -> ModemSnapshot:
        return ModemSnapshot(
            usb_path=modem.usb_path,
            cdc_wdm=modem.cdc_wdm,
        )
